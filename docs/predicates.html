"GoalTest, protocol tests"
testNats

   | g M |
   M := 1000.
   g := [ :x | self nats: x ] asGoal.
   self
      assert: (g solutions next: M)
      equals: (0 to: M - 1)
      modulo: #asOrderedCollection.

   ^ self exportComputationTreeOfGoal: g limitedTo: 10
